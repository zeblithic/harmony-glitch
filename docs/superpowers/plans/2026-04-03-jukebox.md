# Jukebox Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add in-world jukebox entities that play music tracks with distance-based audio falloff, a new Music channel, and a Roxy-ready MusicSource abstraction.

**Architecture:** Jukeboxes are regular entities with optional `playlist` and `audio_radius` fields. Rust owns playback state (JukeboxState), emits continuous JukeboxUpdate AudioEvents per tick. Frontend AudioManager manages Howl instances for active jukeboxes, scaling volume by distance factor. A JukeboxPanel Svelte component shows track list and controls.

**Tech Stack:** Rust (Tauri v2), Svelte 5, TypeScript, Howler.js, CAS pipeline (existing cas-tool)

**Spec:** `docs/superpowers/specs/2026-04-03-jukebox-design.md`

use serde::{Deserialize, Serialize};
use std::collections::HashMap;

/// A single track's metadata from the catalog.
#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(rename_all = "camelCase")]
pub struct TrackDef {
    pub title: String,
    pub artist: String,
    pub duration_secs: f64,
    pub file: String,
}

/// The full track catalog loaded from assets/music/catalog.json.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct TrackCatalog {
    pub tracks: HashMap<String, TrackDef>,
}

/// Per-jukebox-instance runtime state.
#[derive(Debug, Clone)]
pub struct JukeboxState {
    /// Effective playlist (track IDs that exist in catalog).
    pub playlist: Vec<String>,
    pub current_track_index: usize,
    pub playing: bool,
    pub elapsed_secs: f64,
}

impl JukeboxState {
    pub fn new(playlist: Vec<String>) -> Self {
        let playing = !playlist.is_empty();
        Self {
            playlist,
            current_track_index: 0,
            playing,
            elapsed_secs: 0.0,
        }
    }

    /// Advance playback by dt seconds. Auto-advances to next track when current finishes.
    /// Returns the current track ID (if any).
    pub fn tick(&mut self, dt: f64, catalog: &TrackCatalog) -> Option<&str> {
        if !self.playing || self.playlist.is_empty() {
            return self.current_track_id();
        }
        self.elapsed_secs += dt;
        let track_id = &self.playlist[self.current_track_index];
        if let Some(def) = catalog.tracks.get(track_id) {
            if self.elapsed_secs >= def.duration_secs {
                self.elapsed_secs = 0.0;
                self.current_track_index = (self.current_track_index + 1) % self.playlist.len();
            }
        }
        self.current_track_id()
    }

    pub fn current_track_id(&self) -> Option<&str> {
        if self.playlist.is_empty() {
            None
        } else {
            Some(&self.playlist[self.current_track_index])
        }
    }

    pub fn select_track(&mut self, index: usize) {
        if index < self.playlist.len() {
            self.current_track_index = index;
            self.elapsed_secs = 0.0;
            self.playing = true;
        }
    }

    pub fn play(&mut self) {
        if !self.playlist.is_empty() {
            self.playing = true;
        }
    }

    pub fn pause(&mut self) {
        self.playing = false;
    }
}

/// Compute distance factor for spatial audio falloff.
/// Returns None if player is outside audio_radius (no event should be emitted).
/// Returns Some(factor) where factor is 1.0 at the entity and 0.0 at audio_radius.
pub fn distance_factor(player_x: f64, entity_x: f64, audio_radius: f64) -> Option<f64> {
    let distance = (player_x - entity_x).abs();
    if distance >= audio_radius {
        None
    } else {
        Some(1.0 - distance / audio_radius)
    }
}

/// Validate a playlist against the catalog, returning only track IDs that exist.
/// Logs warnings for missing tracks.
pub fn validate_playlist(playlist: &[String], catalog: &TrackCatalog, entity_name: &str) -> Vec<String> {
    let mut valid = Vec::new();
    for track_id in playlist {
        if catalog.tracks.contains_key(track_id) {
            valid.push(track_id.clone());
        } else {
            eprintln!("[jukebox] Entity '{}': unknown track ID '{}', skipping", entity_name, track_id);
        }
    }
    valid
}

/// Parse the track catalog from a JSON string.
pub fn parse_catalog(json: &str) -> Result<TrackCatalog, String> {
    serde_json::from_str(json).map_err(|e| format!("Failed to parse music catalog: {e}"))
}

import { Howl } from 'howler';

export interface MusicSource {
  resolveTrackUrl(trackId: string, filename: string): string;
}

export class LocalMusicSource implements MusicSource {
  resolveTrackUrl(_trackId: string, filename: string): string {
    return `/assets/music/tracks/${filename}`;
  }
}

interface ActiveJukebox {
  entityId: string;
  trackId: string;
  howl: Howl;
  lastSeenTime: number;
}

/** Fade-out duration in ms when a jukebox goes out of range. */
const FADEOUT_MS = 500;

/** Time in ms after which an unseen jukebox is cleaned up. */
const CLEANUP_DELAY_MS = 600;

export interface TrackCatalogEntry {
  title: string;
  artist: string;
  durationSecs: number;
  file: string;
}

export interface TrackCatalog {
  tracks: Record<string, TrackCatalogEntry>;
}

export class JukeboxController {
  private active: Map<string, ActiveJukebox> = new Map();
  private musicSource: MusicSource;
  private catalog: TrackCatalog;
  private getVolume: () => number;

  constructor(musicSource: MusicSource, catalog: TrackCatalog, getVolume: () => number) {
    this.musicSource = musicSource;
    this.catalog = catalog;
    this.getVolume = getVolume;
  }

  /**
   * Process a JukeboxUpdate event from the game loop.
   */
  update(entityId: string, trackId: string, playing: boolean, distanceFactor: number, elapsedSecs: number): void {
    const now = Date.now();
    const existing = this.active.get(entityId);

    if (existing) {
      existing.lastSeenTime = now;

      // Track changed — swap Howl
      if (existing.trackId !== trackId) {
        existing.howl.stop();
        existing.howl.unload();
        const howl = this.createHowl(trackId);
        if (howl) {
          existing.trackId = trackId;
          existing.howl = howl;
          existing.howl.seek(elapsedSecs);
          if (playing) existing.howl.play();
        }
      }

      // Update volume based on distance
      existing.howl.volume(distanceFactor * this.getVolume());

      // Play/pause state
      if (playing && !existing.howl.playing()) {
        existing.howl.play();
      } else if (!playing && existing.howl.playing()) {
        existing.howl.pause();
      }
    } else {
      // New jukebox in range
      const howl = this.createHowl(trackId);
      if (howl) {
        howl.volume(distanceFactor * this.getVolume());
        howl.seek(elapsedSecs);
        if (playing) howl.play();
        this.active.set(entityId, { entityId, trackId, howl, lastSeenTime: now });
      }
    }
  }

  /**
   * Call periodically to clean up jukeboxes that went out of range.
   */
  cleanup(): void {
    const now = Date.now();
    for (const [entityId, jukebox] of this.active) {
      if (now - jukebox.lastSeenTime > CLEANUP_DELAY_MS) {
        jukebox.howl.fade(jukebox.howl.volume(), 0, FADEOUT_MS);
        // Schedule actual cleanup after fade
        setTimeout(() => {
          jukebox.howl.stop();
          jukebox.howl.unload();
        }, FADEOUT_MS);
        this.active.delete(entityId);
      }
    }
  }

  /**
   * Update volume for all active jukeboxes (called when user changes Music volume).
   */
  updateVolume(): void {
    // Volume will be applied on next update() call via distanceFactor * getVolume()
    // For immediate feedback, update all active Howls
    for (const jukebox of this.active.values()) {
      // We don't have the current distanceFactor here, but next tick will correct it.
      // Just skip — the volume adjusts on the next frame anyway.
    }
  }

  dispose(): void {
    for (const jukebox of this.active.values()) {
      jukebox.howl.stop();
      jukebox.howl.unload();
    }
    this.active.clear();
  }

  private createHowl(trackId: string): Howl | null {
    const entry = this.catalog.tracks[trackId];
    if (!entry) return null;
    const url = this.musicSource.resolveTrackUrl(trackId, entry.file);
    return new Howl({
      src: [url],
      preload: true,
      onloaderror: (_id: number, err: unknown) => {
        console.warn(`[JukeboxController] Failed to load track '${trackId}':`, err);
      },
    });
  }
}

  let {
    info,
    visible = false,
    onClose,
    onPlay,
    onPause,
    onSelectTrack,

    info: JukeboxInfo | null;
    visible: boolean;
    onClose?: () => void;
    onPlay?: (entityId: string) => void;
    onPause?: (entityId: string) => void;
    onSelectTrack?: (entityId: string, trackIndex: number) => void;

  let dialogEl: HTMLDialogElement | undefined = $state();
  let previousFocus: HTMLElement | null = null;

  $effect(() => {
    if (visible && dialogEl) {
      if (!dialogEl.open) {
        previousFocus = document.activeElement as HTMLElement | null;
        dialogEl.showModal();
        dialogEl.querySelector<HTMLElement>('button')?.focus();
      }
      return () => {
        if (dialogEl?.open) dialogEl.close();
      };
    } else if (!visible && previousFocus) {
      previousFocus.focus();
      previousFocus = null;
    }
  });

  function handleCancel(e: Event) {
    e.preventDefault();
    onClose?.();
  }

  function formatTime(secs: number): string {
    const m = Math.floor(secs / 60);
    const s = Math.floor(secs % 60);
    return `${m}:${s.toString().padStart(2, '0')}`;
  }

  function togglePlayPause() {
    if (!info) return;
    if (info.playing) {
      onPause?.(info.entityId);
    } else {
      onPlay?.(info.entityId);
    }

    if (!info) return;
    onSelectTrack?.(info.entityId, index);

    if (!info || info.playlist.length === 0) return;
    const prev = (info.currentTrackIndex - 1 + info.playlist.length) % info.playlist.length;
    onSelectTrack?.(info.entityId, prev);
  }

  function nextTrack() {
    if (!info || info.playlist.length === 0) return;
    const next = (info.currentTrackIndex + 1) % info.playlist.length;
    onSelectTrack?.(info.entityId, next);
  }

    info && info.playlist.length > 0 ? info.playlist[info.currentTrackIndex] : null
  );

  let progressPercent = $derived(
    currentTrack && currentTrack.durationSecs > 0
      ? Math.min(100, ((info?.elapsedSecs ?? 0) / currentTrack.durationSecs) * 100)

  );
